Introduction

Fan Triangulation for Convex polygons.

  1. O(n) computation, and O(n) memory.
  2. Very simple algorithm.

Example

taken from

examples/example_geometry_fan_triangulation.cpp

The example demonstrates usage of different linear containers (including the packaged non standard

dynamic_array
).
The example also demonstrates different number types such as fixed point integer(
Q
numbers), float and doubles.
Tessellation computation is only 16 lines of code.

I use the

microgl
project to render the tessellation.
The algorithm output:

  • Will fill The
    output_type
    for you to know how to iterate the triangles (TRIANGLES/FAN/STRIP)
  • Will fill the user supplied
    output_indices
    container with triangulation indices.
  • Will (optionally) fill the user supplied
    output_boundary_info
    container for you to know how to identify the boundary.

24int main() {
25    using number = float;
26    // using number = Q<12>;
27
28    // microgl drawing setup START
29    using Canvas24 = canvas<bitmap<coder::RGB888_PACKED_32>>;
30    Canvas24 canvas(640, 480);
31    sampling::flat_color<> color_red{{255, 0, 0, 255}};
32    // microgl drawing setup END
33
34    auto render_polygon = [&](const container<vertex2<number>> &polygon) {
35
36        // algorithm START
37        // output indices container
38        container<unsigned int> output_indices;
39        // output boundary info container
40        container<boundary_info> output_boundary_buffer;
41        // output triangles indices type
42        microtess::triangles::indices output_triangles_type;
43        // define algorithm
44        using fan = microtess::fan_triangulation<
45                    number, container<unsigned int>, 
46                    container<boundary_info>>;
47        // compute algorithm
48        fan::compute(
49                polygon.data(),
50                polygon.size(),
51                output_indices,
52                &output_boundary_buffer,
53                output_triangles_type
54        );
55        // algorithm END
56
57        // draw triangles batch
58        canvas.clear({255, 255, 255, 255});
59        canvas.drawTriangles<blendmode::Normal, porterduff::FastSourceOverOnOpaque, true>(
60                color_red,
61                matrix_3x3<number>::identity(),
62                polygon.data(),
63                (vertex2<number> *) nullptr,
64                output_indices.data(),
65                output_boundary_buffer.data(),
66                output_indices.size(),
67                output_triangles_type,
68                120);
69
70        // draw triangulation
71        canvas.drawTrianglesWireframe(
72                {0, 0, 0, 255},
73                matrix_3x3<number>::identity(),
74                polygon.data(),
75                output_indices.data(),
76                output_indices.size(),
77                output_triangles_type,
78                255);
79    };
80
81    auto render = [&](void*, void*, void*) -> void {
82        render_polygon(poly_diamond<number>());
83    };
84
85    example_run(&canvas, render);
